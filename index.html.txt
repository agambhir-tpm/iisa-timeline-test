  IISA EPIC TIMELINE  —  Confluence embed (iframe)
  ═══════════════════════════════════════════════════════════════════

  WHY IFRAME INSTEAD OF FETCH
  ───────────────────────────
  fetch() fails ("Failed to fetch") because GAS web apps deployed
  with org-restricted access redirect to Google login — and that
  redirect has no CORS headers, so the browser blocks it.

  The iframe approach avoids this entirely:
    • The browser loads the full GAS web app URL inside the frame.
    • google.script.run works because GAS serves the HTML itself.
    • No CORS, no auth redirect issues.

  SETUP  (2 steps)
  ────────────────
  1. Replace PASTE_YOUR_GAS_URL_HERE below with your Web App URL:
       https://script.google.com/macros/s/AKfycb.../exec

  2. In Confluence, add an HTML macro and paste this whole file.
     (Confluence Server/DC: built-in HTML macro)
     (Confluence Cloud: use "HTML for Confluence" or "Advanced Macro" app,
      or ask your admin to allowlist script.google.com for iframes.)
  ═══════════════════════════════════════════════════════════════════
-->

<script>
(function () {
  var frame   = document.getElementById('iisa-frame');
  var loading = document.getElementById('iisa-loading');
  var error   = document.getElementById('iisa-error');
  var openLink = document.getElementById('iisa-open-link');

  // Mirror GAS URL to the fallback "open in new tab" link
  openLink.href = frame.src;

  function showError() {
    loading.style.display = 'none';
    frame.style.display   = 'none';
    error.style.display   = 'flex';
  }

  frame.addEventListener('load', function () {
    loading.style.display = 'none';

    // Cross-origin check: if we can read location it's not GAS (login redirect)
    try {
      var loc = frame.contentWindow.location.href;
      if (loc.indexOf('accounts.google.com') !== -1) showError();
    } catch (e) {
      // Can't read location = cross-origin = GAS loaded correctly ✓
    }
  });

  frame.addEventListener('error', showError);

  // Safety net: if iframe hasn't loaded after 60 s, show error
  var guard = setTimeout(showError, 60000);
  frame.addEventListener('load', function () { clearTimeout(guard); });
})();
</script>

(function() {

var WEEK_W      = 72;
var LABEL_W     = 180;
var BAR_H       = 26;
var BAR_GAP     = 5;
var ROW_PAD_TOP = 8;
var ROW_PAD_BOT = 8;

var BACKLOG_COLOR = {
  'Integrations Backlog': '#0891b2',
  'AI Agentic Backlog':   '#7c3aed'
};

var MONTHS_LONG = ['January','February','March','April','May','June',
                   'July','August','September','October','November','December'];

var iisaAllEpics   = [];
var iisaEpicTipMap = {};
var iisaLoadTimer  = null;
var iisaRagFieldPresent = false;  // set to true when backend returns ragFieldId

/* ── Date helpers ─────────────────────────────────────────────── */
function todayDate() {
  var d = new Date(); d.setHours(0,0,0,0); return d;
}
function getMonday(d) {
  var dt = new Date(d), day = dt.getDay(), off = day===0?-6:1-day;
  dt.setDate(dt.getDate()+off); dt.setHours(0,0,0,0); return dt;
}
function addDays(d,n)   { var dt=new Date(d); dt.setDate(dt.getDate()+n); return dt; }
function addMonths(d,n) { var dt=new Date(d); dt.setMonth(dt.getMonth()+n); dt.setHours(0,0,0,0); return dt; }
function fmtDate(d)     { return d ? d.toLocaleDateString('en-US',{month:'short',day:'numeric',year:'numeric'}) : '—'; }
function esc(s) {
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

/* ── Data fetch via fetch() → GAS JSON endpoint ──────────────── */
window.iisaRefreshData = function() {
  iisaShowLoading(true);
  iisaSetRefreshState(true);
  iisaClearMessages();

  // Timeout guard — 50 s
  iisaLoadTimer = setTimeout(function() {
    iisaShowLoading(false);
    iisaSetRefreshState(false);
    iisaShowMsg('error',
      'Request timed out (50 s). Check that GAS_URL is correct, the web app is deployed ' +
      'with "Anyone" access, and JIRA credentials are set in Script Properties.');
    iisaRenderEmpty('Timed out.');
  }, 50000);

  fetch(GAS_URL + '?action=getEpics', { redirect: 'follow' })
    .then(function(resp) {
      if (!resp.ok) throw new Error('HTTP ' + resp.status);
      return resp.json();
    })
    .then(iisaOnDataLoaded)
    .catch(function(err) {
      iisaOnDataError({ message: err.message || String(err) });
    });
};

function iisaOnDataLoaded(result) {
  clearTimeout(iisaLoadTimer);
  iisaShowLoading(false);
  iisaSetRefreshState(false);

  if (!result) {
    iisaShowMsg('error', 'Server returned no data.');
    iisaRenderEmpty('No data returned.');
    return;
  }
  if (result.warnings && result.warnings.length) {
    result.warnings.forEach(function(w) { iisaShowMsg('warn', w); });
  }
  iisaRagFieldPresent = !!result.ragFieldId;
  iisaAllEpics = iisaProcessEpics(result.epics || []);
  iisaPopulateFilterDropdowns();
  iisaApplyFilters();
}

function iisaOnDataError(err) {
  clearTimeout(iisaLoadTimer);
  iisaShowLoading(false);
  iisaSetRefreshState(false);
  iisaShowMsg('error', 'Could not reach GAS endpoint: ' + (err.message || String(err)));
  iisaRenderEmpty('Connection failed — check GAS_URL and deployment settings.');
}

/* ── Process epics ────────────────────────────────────────────── */
function iisaProcessEpics(raw) {
  var now = todayDate();
  return raw.map(function(e) {
    var endDate, dueDateRaw = null;
    if (e.dueDate) {
      var p = e.dueDate.split('-');
      endDate = new Date(+p[0], +p[1]-1, +p[2]);
      dueDateRaw = endDate;
    } else {
      endDate = addDays(now, 84);
    }
    return {
      key:        e.key,
      summary:    e.summary || '(no summary)',
      assignee:   e.assignee || 'Unassigned',
      status:     e.status   || 'Unknown',
      labels:     Array.isArray(e.labels) ? e.labels : [],
      backlog:    e.backlog  || '',
      startDate:  addDays(endDate, -28),
      endDate:    endDate,
      dueDateRaw: dueDateRaw
    };
  });
}

/* ── Filters ──────────────────────────────────────────────────── */
function iisaPopulateFilterDropdowns() {
  var statuses = [], labels = [], ragValues = [];
  iisaAllEpics.forEach(function(e) {
    if (e.status && statuses.indexOf(e.status)===-1) statuses.push(e.status);
    e.labels.forEach(function(l){ if(l && labels.indexOf(l)===-1) labels.push(l); });
    if (e.ragStatus && ragValues.indexOf(e.ragStatus)===-1) ragValues.push(e.ragStatus);
  });
  statuses.sort(); labels.sort();
  // Sort RAG: Red → Amber → Green → others
  var ragOrder = ['red','amber','green'];
  ragValues.sort(function(a,b){
    var ai=ragOrder.indexOf(a.toLowerCase()), bi=ragOrder.indexOf(b.toLowerCase());
    if(ai===-1) ai=99; if(bi===-1) bi=99;
    return ai-bi;
  });

  iisaBuildDropdown('status-dd','status-lbl',statuses,'All statuses');
  iisaBuildDropdown('labels-dd','labels-lbl',labels,'All labels');
  iisaBuildRagDropdown('rag-dd','rag-lbl',ragValues);
}

function iisaBuildRagDropdown(ddId, lblId, items) {
  var dd = document.getElementById(ddId);
  if (!items.length) {
    dd.innerHTML = '<div class="ms-empty">' + (iisaRagFieldPresent ? 'No RAG values found' : 'RAG field not configured') + '</div>';
    return;
  }
  var html = '';
  items.forEach(function(item) {
    var sid     = ddId+'-'+item.replace(/[^a-z0-9]/gi,'_');
    var dotCls  = iisaRagCssClass(item);
    html += '<div class="ms-item">'
          + '<input type="checkbox" id="'+esc(sid)+'" value="'+esc(item)+'"'
          + ' onchange="iisaSyncLabel(\''+esc(ddId)+'\',\''+esc(lblId)+'\',\'All RAG\');iisaApplyFilters()">'
          + '<span class="rag-dot '+dotCls+'"></span>'
          + '<label for="'+esc(sid)+'">'+esc(item)+'</label>'
          + '</div>';
  });
  dd.innerHTML = html;
}

function iisaBuildDropdown(ddId, lblId, items, placeholder) {
  var dd = document.getElementById(ddId);
  if (!items.length) { dd.innerHTML='<div class="ms-empty">None available</div>'; return; }
  var html = '';
  items.forEach(function(item) {
    var sid = ddId+'-'+item.replace(/[^a-z0-9]/gi,'_');
    html += '<div class="ms-item">'
          + '<input type="checkbox" id="'+esc(sid)+'" value="'+esc(item)+'"'
          + ' onchange="iisaSyncLabel(\''+esc(ddId)+'\',\''+esc(lblId)+'\',\''+esc(placeholder)+'\');iisaApplyFilters()">'
          + '<label for="'+esc(sid)+'">'+esc(item)+'</label></div>';
  });
  dd.innerHTML = html;
}

window.iisaSyncLabel = function(ddId, lblId, placeholder) {
  var checked = iisaGetChecked(ddId);
  var el = document.getElementById(lblId);
  if (!el) return;
  el.textContent = !checked.length ? placeholder : checked.length===1 ? checked[0] : checked.length+' selected';
};

function iisaGetChecked(ddId) {
  var out = [];
  document.getElementById(ddId).querySelectorAll('input:checked').forEach(function(cb){ out.push(cb.value); });
  return out;
}

window.iisaApplyFilters = function() {
  var showInt = document.getElementById('cb-integrations').checked;
  var showAI  = document.getElementById('cb-agentic').checked;
  var selStat = iisaGetChecked('status-dd');
  var selLbl  = iisaGetChecked('labels-dd');
  var selRag  = iisaGetChecked('rag-dd');

  var filtered = iisaAllEpics.filter(function(e) {
    if (e.backlog==='Integrations Backlog' && !showInt) return false;
    if (e.backlog==='AI Agentic Backlog'   && !showAI)  return false;
    if (selStat.length && selStat.indexOf(e.status)===-1) return false;
    if (selLbl.length) {
      if (!e.labels.some(function(l){ return selLbl.indexOf(l)!==-1; })) return false;
    }
    if (selRag.length) {
      // Match on ragStatus; treat null/undefined as "Not set" for filtering
      var rag = e.ragStatus || 'Not set';
      if (selRag.indexOf(rag)===-1) return false;
    }
    return true;
  });

  filtered.length ? iisaBuildTimeline(filtered) : iisaRenderEmpty('No epics match the current filters.');
};

/* ── Timeline ─────────────────────────────────────────────────── */
function iisaBuildTimeline(epics) {
  var now      = todayDate();
  var tlStart  = getMonday(addMonths(now, -6));
  var minEnd   = addMonths(now, 6);
  var maxEnd   = epics.reduce(function(mx,e){ return e.endDate>mx?e.endDate:mx; }, minEnd);
  var tlEnd    = getMonday(addDays(maxEnd, 14));
  var totalWeeks = Math.round((tlEnd-tlStart)/(7*86400000));
  var totalWidth = totalWeeks * WEEK_W;

  var weeks = [];
  for (var w=0; w<totalWeeks; w++) {
    var ws = addDays(tlStart, w*7), we = addDays(ws,7);
    weeks.push({ index:w, date:ws, x:w*WEEK_W, isCurrent:ws<=now&&we>now,
                 label:(ws.getMonth()+1)+'/'+ws.getDate() });
  }

  var months = [];
  weeks.forEach(function(wk) {
    var key = wk.date.getFullYear()+'-'+wk.date.getMonth();
    var last = months[months.length-1];
    if (!last||last.key!==key) months.push({key:key,name:MONTHS_LONG[wk.date.getMonth()].substring(0,3)+' '+wk.date.getFullYear(),x:wk.x,width:WEEK_W});
    else last.width+=WEEK_W;
  });

  var cwWeek = weeks.filter(function(wk){return wk.isCurrent;})[0];
  var cwX    = cwWeek ? cwWeek.x : -1;
  var todayX = Math.round((now-tlStart)/86400000*WEEK_W/7);

  var assigneeMap = {};
  epics.forEach(function(e){ if(!assigneeMap[e.assignee]) assigneeMap[e.assignee]=[]; assigneeMap[e.assignee].push(e); });
  var assignees = Object.keys(assigneeMap).filter(function(a){return a!=='Unassigned';}).sort();
  if (assigneeMap['Unassigned']) assignees.push('Unassigned');

  var rowData = assignees.map(function(assignee) {
    var aEpics = assigneeMap[assignee].slice().sort(function(a,b){return a.startDate-b.startDate;});
    var laneEnds = [];
    aEpics.forEach(function(epic) {
      var lane=-1;
      for(var l=0;l<laneEnds.length;l++){ if(epic.startDate>=laneEnds[l]){lane=l;break;} }
      if(lane===-1){lane=laneEnds.length;laneEnds.push(null);}
      laneEnds[lane]=epic.endDate; epic._lane=lane;
    });
    var numLanes=Math.max(1,laneEnds.length);
    return {assignee:assignee, epics:aEpics, rowHeight:ROW_PAD_TOP+numLanes*(BAR_H+BAR_GAP)-BAR_GAP+ROW_PAD_BOT};
  });

  iisaEpicTipMap = {};
  epics.forEach(function(e){ iisaEpicTipMap[e.key]=e; });

  var html = '';
  // Header
  html += '<div id="tl-header"><div class="tl-corner"><div class="corner-top">Month</div><div class="corner-bot">Week of</div></div>'
        + '<div id="tl-header-canvas" style="width:'+totalWidth+'px">';
  html += '<div class="month-row">';
  months.forEach(function(m){ html+='<div class="month-cell" style="width:'+m.width+'px">'+esc(m.name)+'</div>'; });
  html += '</div><div class="week-row">';
  weeks.forEach(function(wk){ html+='<div class="week-cell'+(wk.isCurrent?' is-current':'')+'" style="width:'+WEEK_W+'px">'+esc(wk.label)+'</div>'; });
  html += '</div></div></div>';
  // Body
  html += '<div id="tl-body">';
  rowData.forEach(function(row) {
    var isU = row.assignee==='Unassigned';
    html += '<div class="tl-row">'
          + '<div class="row-label'+(isU?' unassigned':'')+'" style="height:'+row.rowHeight+'px">'+esc(row.assignee)+'</div>'
          + '<div class="row-canvas" style="width:'+totalWidth+'px;height:'+row.rowHeight+'px;'
          + 'background-image:repeating-linear-gradient(to right,transparent,transparent '+(WEEK_W-1)+'px,#e8edf2 '+(WEEK_W-1)+'px,#e8edf2 '+WEEK_W+'px)">';
    if (cwX>=0) html+='<div class="cw-band" style="left:'+cwX+'px;width:'+WEEK_W+'px"></div>';
    html+='<div class="today-line" style="left:'+todayX+'px"></div>';
    row.epics.forEach(function(epic) {
      var x=Math.round((epic.startDate-tlStart)/86400000*WEEK_W/7);
      var bw=Math.max(Math.round((epic.endDate-epic.startDate)/86400000*WEEK_W/7),32);
      var y=ROW_PAD_TOP+epic._lane*(BAR_H+BAR_GAP);
      var color=BACKLOG_COLOR[epic.backlog]||'#6366f1';
      var sc=iisaStatusClass(epic.status);
      html+='<div class="epic-bar '+sc+'" style="left:'+x+'px;top:'+y+'px;width:'+bw+'px;height:'+BAR_H+'px;background:'+color+'"'
           +' data-key="'+esc(epic.key)+'"'
           +' onmouseenter="iisaShowTip(event,this)" onmouseleave="iisaHideTip()" onmousemove="iisaMoveTip(event)">'
           +'<span class="bar-text">'+esc(epic.summary)+'</span></div>';
    });
    html+='</div></div>';
  });
  html+='</div>';

  document.getElementById('timeline-inner').innerHTML = html;

  requestAnimationFrame(function() {
    var c = document.getElementById('timeline-container');
    c.scrollLeft = Math.max(0, todayX - (LABEL_W + 3*WEEK_W));
  });
}

function iisaStatusClass(status) {
  var s=(status||'').toLowerCase().trim();
  if(['to do','todo','open','backlog','new'].indexOf(s)!==-1) return 'st-todo';
  if(['in progress','in development','in review','in-progress','development','review','in dev'].indexOf(s)!==-1) return 'st-inprog';
  if(['done','closed','resolved','complete','completed','released'].indexOf(s)!==-1) return 'st-done';
  if(s==='blocked') return 'st-blocked';
  return 'st-other';
}

  var x=e.clientX+15, y=e.clientY+15;

  if(y+220>window.innerHeight) y=e.clientY-230;

};

/* ── RAG helpers ──────────────────────────────────────────────── */
function iisaRagHexColor(ragStatus) {
  if (!ragStatus) return '#94a3b8';
  var s = ragStatus.toLowerCase().trim();
  if (s === 'red'   || s === 'r') return '#ef4444';
  if (s === 'amber' || s === 'a' || s === 'orange') return '#f59e0b';
  if (s === 'green' || s === 'g') return '#22c55e';
  return '#94a3b8';
}

function iisaRagCssClass(ragStatus) {
  if (!ragStatus) return 'rag-grey';
  var s = ragStatus.toLowerCase().trim();
  if (s === 'red'   || s === 'r') return 'rag-red';
  if (s === 'amber' || s === 'a' || s === 'orange') return 'rag-amber';
  if (s === 'green' || s === 'g') return 'rag-green';
  return 'rag-grey';
}

});
